    import pandas as pd
    from pathlib import Path
    from reports.examen_de_eje.generator import (
        ExamenDeEjeGenerator, ExamenPlan, UnitStats
    )

    # Build a synthetic analysis result using real unit names from the bank
    bank = pd.read_excel("inputs/M30M2-EXAMEN DE EJE 1-DATA.xlsx")
    bank.columns = [c.strip().lower() for c in bank.columns]
    unit_names = list(dict.fromkeys(bank["unidad"].dropna().tolist()))
    print("Units found in bank:", unit_names)

    plan = ExamenPlan(
        assessment_type="M30M2",
        student_id="smoke-test-user",
        email="smoke@example.com",
        assessment_name="M30M2-EXAMEN DE EJE 1-DATA",
        unit_order=unit_names,
    )
    # Assign synthetic scores: first unit <50%, second 50-80%, third >=80%
    scores = [(3, 7), (5, 8), (9, 10)]
    for i, name in enumerate(unit_names):
        correct, total = scores[i] if i < len(scores) else (5, 10)
        plan.units[name] = UnitStats(name=name, total=total, correct=correct)

    gen = ExamenDeEjeGenerator()
    output_dir = gen.render({("M30M2", "smoke@example.com"): plan})
    pdfs = list(output_dir.glob("*.pdf"))
    print(f"PDFs generated: {[p.name for p in pdfs]}")
    assert len(pdfs) == 1, f"Expected 1 PDF, got {len(pdfs)}"
    print("Smoke test PASSED:", pdfs[0])
